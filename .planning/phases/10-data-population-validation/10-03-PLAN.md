```python
class BaseStateScraper(abc.ABC):
    source_name: str  # e.g. "state_california"
    state_code: str   # e.g. "ca"

    @abc.abstractmethod
    def fetch_records(self) -> list[dict]:
        """Fetch raw records from the state portal."""

    @abc.abstractmethod
    def normalize_record(self, raw: dict) -> dict | None:
        """Normalize a raw record into Opportunity-compatible dict."""

    def run(self, session=None) -> dict:
        """Orchestrate fetch + normalize + upsert. Returns stats dict."""
```
