from grantflow.database import SessionLocal
from sqlalchemy import text
import os, subprocess, time, json

s = SessionLocal()

# Check multi-source
sources = dict(s.execute(text('SELECT source, COUNT(*) FROM opportunities GROUP BY source')).fetchall())
print(f'Sources: {list(sources.keys())}')
assert len(sources) >= 3, f'Only {len(sources)} sources, need 3+'
assert sources.get('grants_gov', 0) > 0, 'No Grants.gov data'

# Check normalization
bare = s.execute(text(\"SELECT COUNT(*) FROM opportunities WHERE eligible_applicants IS NOT NULL AND eligible_applicants NOT LIKE '[%' AND eligible_applicants != ''\")).scalar()
assert bare == 0, f'{bare} bare eligibility codes remain'

raw_cat = s.execute(text(\"SELECT COUNT(*) FROM opportunities WHERE category IS NOT NULL AND LENGTH(category) <= 2\")).scalar()
assert raw_cat == 0, f'{raw_cat} raw category codes remain'

# Check topic tags (only if OPENAI_API_KEY was set)
if os.getenv('OPENAI_API_KEY'):
    tagged = s.execute(text(\"SELECT COUNT(*) FROM opportunities WHERE topic_tags IS NOT NULL AND topic_tags != ''\")).scalar()
    assert tagged > 0, 'No topic tags despite OPENAI_API_KEY being set'
    print(f'Topic tags: {tagged}')

s.close()

# Live API smoke test
proc = subprocess.Popen(
    ['uv', 'run', 'uvicorn', 'grantflow.api.app:app', '--host', '127.0.0.1', '--port', '8199'],
    stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL
)
time.sleep(4)
try:

finally:
    proc.terminate()

print('PASS: all validation checks passed')
